% Custom commands for subheadings
\newcommand{\subheading}[5]{
    \ifx&#2&%
    \textbf{#1} \hfill \textbf{#3} \\
    \textit{\small #4} \hfill \textit{\small #5} \\
    \else
    \textbf{#1}, \emph{#2} \hfill \textbf{#3} \\
    \textit{\small #4} \hfill \textit{\small #5} \\
    \fi
}

\newcommand{\subheadingLine}[3]{
    \ifx&#2&%
    \textbf{#1} \hfill \textbf{#3} \\
    \else
    \textbf{#1}, \emph{#2} \hfill \textbf{#3} \\
    \fi
}
